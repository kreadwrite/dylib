    16→    if (text.length == 0) return;
    17→    NSString *encoded = [text stringByAddingPercentEncodingWithAllowedCharacters:[NSCharacterSet URLQueryAllowedCharacterSet]];
    18→    // sl=auto detects the source language automatically; change tl=ru to your preferred target language
    19→    NSString *urlString = [NSString stringWithFormat:@"https://translate.google.com/?sl=auto&tl=ru&text=%@&op=translate", encoded];
    20→    NSURL *url = [NSURL URLWithString:urlString];
    21→    if (url) {
    22→        [[UIApplication sharedApplication] openURL:url options:@{} completionHandler:nil];
    23→    }

    26→// Ghost: schedules the clipboard to be wiped a short while after PXTok copies
    27→// something into it, so a copied link/description/comment doesn't linger

    31→    if (![BHIManager ghostClipboardClear]) return;
    32→    NSString *snapshot = [UIPasteboard generalPasteboard].string;

    36→        // don't stomp on something the user copied from elsewhere in the meantime.

    38→            [UIPasteboard generalPasteboard].string = @"";

   232→      double totalSeconds = [[NSUserDefaults standardUserDefaults] doubleForKey:@"px_total_seconds"];
   233→      [[NSUserDefaults standardUserDefaults] setDouble:totalSeconds + elapsed forKey:@"px_total_seconds"];
   234→      [[NSUserDefaults standardUserDefaults] setDouble:0 forKey:@"px_session_start"];

   241→      double todaySeconds = [[NSUserDefaults standardUserDefaults] doubleForKey:@"px_today_seconds"];
   242→      if (![storedDay isEqualToString:todayKey]) {
   243→          todaySeconds = 0;
   244→          [[NSUserDefaults standardUserDefaults] setInteger:0 forKey:@"px_today_videos"];
   245→          [[NSUserDefaults standardUserDefaults] setValue:todayKey forKey:@"px_stats_day"];
   246→      }
   247→      [[NSUserDefaults standardUserDefaults] setDouble:todaySeconds + elapsed forKey:@"px_today_seconds"];
   248→  }
