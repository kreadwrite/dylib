    23→// Ghost: schedules the clipboard to be wiped a short while after PXTok copies
    24→// something into it, so a copied link/description/comment doesn't linger
    25→// indefinitely in your clipboard (visible to any app that reads it, including
    26→// the system clipboard-access popup on iOS 14+).
    27→static void PXScheduleClipboardClear(void) {
    28→    if (![BHIManager ghostClipboardClear]) return;
    29→    NSString *snapshot = [UIPasteboard generalPasteboard].string;
    30→    double delay = 30.0;
    31→    dispatch_after(dispatch_time(DISPATCH_TIME_NOW, (int64_t)(delay * NSEC_PER_SEC)), dispatch_get_main_queue(), ^{
    32→        // Only clear if the clipboard still holds exactly what we put there —
    33→        // don't stomp on something the user copied from elsewhere in the meantime.
    34→        if ([[UIPasteboard generalPasteboard].string isEqualToString:snapshot]) {
    35→            [UIPasteboard generalPasteboard].string = @"";
    36→        }
    37→    });
