    14→static void PXTranslateText(NSString *text) {
    15→    if (text.length == 0) return;
    16→    NSString *encoded = [text stringByAddingPercentEncodingWithAllowedCharacters:[NSCharacterSet URLQueryAllowedCharacterSet]];
    17→    // sl=auto detects the source language automatically; change tl=ru to your preferred target language
    18→    NSString *urlString = [NSString stringWithFormat:@"https://translate.google.com/?sl=auto&tl=ru&text=%@&op=translate", encoded];
    19→    NSURL *url = [NSURL URLWithString:urlString];
    20→    if (url) {
    21→        [[UIApplication sharedApplication] openURL:url options:@{} completionHandler:nil];
    22→    }
    23→}

    25→// Ghost: schedules the clipboard to be wiped a short while after PXTok copies
    26→// something into it, so a copied link/description/comment doesn't linger
    27→// indefinitely in your clipboard (visible to any app that reads it, including
    28→// the system clipboard-access popup on iOS 14+).
    29→static void PXScheduleClipboardClear(void) {
    30→    if (![BHIManager ghostClipboardClear]) return;
    31→    NSString *snapshot = [UIPasteboard generalPasteboard].string;
    32→    double delay = 30.0;
    33→    dispatch_after(dispatch_time(DISPATCH_TIME_NOW, (int64_t)(delay * NSEC_PER_SEC)), dispatch_get_main_queue(), ^{
    34→        // Only clear if the clipboard still holds exactly what we put there —
    35→        // don't stomp on something the user copied from elsewhere in the meantime.
    36→        if ([[UIPasteboard generalPasteboard].string isEqualToString:snapshot]) {
    37→            [UIPasteboard generalPasteboard].string = @"";
    38→        }
    39→    });
    40→}

   207→  double start = [[NSUserDefaults standardUserDefaults] doubleForKey:@"px_session_start"];

   209→      double elapsed = [[NSDate date] timeIntervalSince1970] - start;
   210→      double totalSeconds = [[NSUserDefaults standardUserDefaults] doubleForKey:@"px_total_seconds"];
   211→      [[NSUserDefaults standardUserDefaults] setDouble:totalSeconds + elapsed forKey:@"px_total_seconds"];
   212→      [[NSUserDefaults standardUserDefaults] setDouble:0 forKey:@"px_session_start"];
   213→
   214→      // Also fold into "today"'s bucket, resetting if the day rolled over.
   215→      NSDateFormatter *formatter = [[NSDateFormatter alloc] init];
   216→      formatter.dateFormat = @"yyyy-MM-dd";
   217→      NSString *todayKey = [formatter stringFromDate:[NSDate date]];
   218→      NSString *storedDay = [[NSUserDefaults standardUserDefaults] stringForKey:@"px_stats_day"];
   219→      double todaySeconds = [[NSUserDefaults standardUserDefaults] doubleForKey:@"px_today_seconds"];
   220→      if (![storedDay isEqualToString:todayKey]) {
   221→          todaySeconds = 0;
   222→          [[NSUserDefaults standardUserDefaults] setInteger:0 forKey:@"px_today_videos"];
   223→          [[NSUserDefaults standardUserDefaults] setValue:todayKey forKey:@"px_stats_day"];
   224→      }
   225→      [[NSUserDefaults standardUserDefaults] setDouble:todaySeconds + elapsed forKey:@"px_today_seconds"];
   226→  }
